### 1. Create a Client

Set up a [Public Client](/docs/clients/public/) with a [Transport](/docs/clients/transports/http/) and optional [Chain](/docs/clients/intro/) configuration.

<Tabs>
  <TabItem label="Go">
```go
package main

import (
    "context"
    "fmt"
    "log"
    
    "github.com/ChefBingbong/viem-go/client"
    "github.com/ChefBingbong/viem-go/client/transport"
    "github.com/ChefBingbong/viem-go/chain/definitions"
)

func main() {
    // Create a public client
    publicClient, err := client.CreatePublicClient(client.PublicClientConfig{
        Chain:     definitions.Mainnet,
        Transport: transport.HTTP("https://eth.llamarpc.com"),
    })
    if err != nil {
        log.Fatal(err)
    }
    defer publicClient.Close()
    
    fmt.Println("Client created successfully!")
}
```
  </TabItem>
  <TabItem label="TypeScript (viem)">
```typescript
import { createPublicClient, http } from 'viem'
import { mainnet } from 'viem/chains'

const publicClient = createPublicClient({
  chain: mainnet,
  transport: http(),
})

console.log('Client created successfully!')
```
  </TabItem>
</Tabs>

<Aside type="tip">
In production, always use an authenticated RPC endpoint (Infura, Alchemy, etc.) instead of public endpoints.
</Aside>

### 2. Read Blockchain Data

Once you have a client, you can query the blockchain:

<Tabs>
  <TabItem label="Go">
```go
// Get the current block number
blockNumber, err := publicClient.GetBlockNumber(context.Background())
if err != nil {
    log.Fatal(err)
}
fmt.Println("Block number:", blockNumber)

// Get an account balance
balance, err := publicClient.GetBalance(
    context.Background(),
    common.HexToAddress("0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045"),
)
if err != nil {
    log.Fatal(err)
}
fmt.Println("Balance:", balance)
```
  </TabItem>
  <TabItem label="TypeScript (viem)">
```typescript
// Get the current block number
const blockNumber = await publicClient.getBlockNumber()
console.log('Block number:', blockNumber)

// Get an account balance
const balance = await publicClient.getBalance({
  address: '0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045',
})
console.log('Balance:', balance)
```
  </TabItem>
</Tabs>

### 3. Read from a Contract

Use `ReadContract` to call view/pure functions on smart contracts:

<Tabs>
  <TabItem label="Go">
```go
import (
    "github.com/ethereum/go-ethereum/common"
    "math/big"
)

// Read USDC balance
result, err := publicClient.ReadContract(context.Background(), client.ReadContractOptions{
    Address:      common.HexToAddress("0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48"),
    ABI:          `[{"name":"balanceOf","type":"function","inputs":[{"name":"owner","type":"address"}],"outputs":[{"type":"uint256"}]}]`,
    FunctionName: "balanceOf",
    Args:         []any{common.HexToAddress("0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045")},
})
if err != nil {
    log.Fatal(err)
}

balance := result.(*big.Int)
fmt.Println("USDC Balance:", balance)
```
  </TabItem>
  <TabItem label="TypeScript (viem)">
```typescript
import { parseAbi } from 'viem'

// Read USDC balance
const balance = await publicClient.readContract({
  address: '0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48',
  abi: parseAbi(['function balanceOf(address) view returns (uint256)']),
  functionName: 'balanceOf',
  args: ['0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045'],
})

console.log('USDC Balance:', balance)
```
  </TabItem>
</Tabs>

### 4. Using Pre-built Contract Bindings

For common contracts like ERC20, use the typed bindings:

<Tabs>
  <TabItem label="Go">
```go
import "github.com/ChefBingbong/viem-go/contracts/erc20"

// Create typed ERC20 instance
usdc := common.HexToAddress("0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48")
token, err := erc20.New(usdc, publicClient)
if err != nil {
    log.Fatal(err)
}

// Call typed methods
balance, err := token.BalanceOf(
    context.Background(),
    common.HexToAddress("0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045"),
)
fmt.Println("Balance:", balance)

// Get token metadata
name, _ := token.Name(context.Background())
symbol, _ := token.Symbol(context.Background())
decimals, _ := token.Decimals(context.Background())

fmt.Printf("Token: %s (%s) - %d decimals\n", name, symbol, decimals)
```
  </TabItem>
  <TabItem label="TypeScript (viem)">
```typescript
import { erc20Abi } from 'viem'

const usdc = '0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48'

// Read using the standard ERC20 ABI
const balance = await publicClient.readContract({
  address: usdc,
  abi: erc20Abi,
  functionName: 'balanceOf',
  args: ['0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045'],
})

const name = await publicClient.readContract({
  address: usdc,
  abi: erc20Abi,
  functionName: 'name',
})

console.log('Balance:', balance)
console.log('Name:', name)
```
