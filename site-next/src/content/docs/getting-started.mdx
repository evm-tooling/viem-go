### 1. Create a Client

Set up a [Public Client](/docs/clients/public/) with a [Transport](/docs/clients/transports/http/) and optional [Chain](/docs/clients/intro/) configuration.

<CodeGroup 

  tabs={[
    {
      title: "Go",
      language: "go",
      code: `package main

import (
    "context"
    "fmt"
    "log"
    
    "github.com/ChefBingbong/viem-go/client"
    "github.com/ChefBingbong/viem-go/client/transport"
    "github.com/ChefBingbong/viem-go/chain/definitions"
)

func main() {
    // Create a public client
    publicClient, err := client.CreatePublicClient(client.PublicClientConfig{ // [!code focus:3]
        Chain:     definitions.Mainnet,
        Transport: transport.HTTP("https://eth.llamarpc.com"),
    })
    if err != nil {
        log.Fatal(err)
    }
    defer publicClient.Close()
    
    fmt.Println("Client created successfully!")
}`
    },
    {
      title: "viem",
      language: "ts",
      code: `import { createPublicClient, http } from 'viem'
import { mainnet } from 'viem/chains'

const publicClient = createPublicClient({ // [!code focus:3]
  chain: mainnet,
  transport: http(),
})

console.log('Client created successfully!')`
    }
  ]}
/>

<Aside type="note">
In production, always use an authenticated RPC endpoint (Infura, Alchemy, etc.) instead of public endpoints
</Aside>


### 2. Read Blockchain Data

Once you have a client, you can query the blockchain:

<CodeGroup 

  tabs={[
    {
      title: "Go",
      language: "go",
      code: `// Get the current block number
blockNumber, err := publicClient.GetBlockNumber(context.Background()) // [!code focus]
if err != nil {
    log.Fatal(err)
}
fmt.Println("Block number:", blockNumber)

// Read USDC balance
result, err := publicClient.ReadContract(context.Background(), client.ReadContractOptions{ // [!code focus:5]
    Address:      common.HexToAddress("0xA0b86991c6218b36c1d19D4a2e9Eb0cE3606eB48"),
    ABI:          \`[{"name":"balanceOf","type":"function","inputs":[{"name":"owner","type":"address"}],"outputs":[{"type":"uint256"}]}]\`,
    FunctionName: "balanceOf",
    Args:         []any{common.HexToAddress("0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045")},
})
if err != nil {
    log.Fatal(err)
}

balance := result.(*big.Int)
fmt.Println("USDC Balance:", balance)`
    },
    {
      title: "viem",
      language: "ts",
      code: `// Get the current block number
const blockNumber = await publicClient.getBlockNumber()
console.log('Block number:', blockNumber)

// Get an account balance
const balance = await publicClient.getBalance({
  address: '0xd8dA6BF26964aF9D7eEd9e03E53415D37aA96045',
})
console.log('Balance:', balance)`
    }
  ]}
/>

<Aside type="tip">
<span><code>ReadContract</code> works with any contract function, not just <code>balanceOf</code>. You can call any ERC-20, ERC-721, or ERC-1155 method -- or any custom contract ABI -- using the same pattern.</span>
</Aside>
